    fun setupNettyClient(ip: String, port: Int, isHeartBeat: Boolean, commuForm: CommuForm): NettyTcpClient {
        return NettyTcpClient.Builder()
                .setHost(ip)                    //设置服务端地址
                .setTcpPort(port)       //设置服务端端口号
                .setMaxReconnectTimes(5000)     //设置最大重连次数
                .setReconnectIntervalTime(30000)//设置重连间隔时间。单位：秒
                .setSendheartBeat(isHeartBeat)         //设置发送心跳
                .setHeartBeatInterval(30)       //设置心跳间隔时间。单位：秒
                .setHeartBeatData(Json.encodeToString(commuForm)) //设置心跳数据，可以是String类型，也可以是byte[]，以后设置的为准
                .setIndex(0)                    //设置客户端标识.(因为可能存在多个tcp连接)
                .build()
    }
    fun init(isheartBeat: Boolean) {
        Log.d(TAG, " init")

        pSharedPrefs = sharedPrefs
        ip = getStringPref(sharedPrefs, "IP")
        port = getIntPref(sharedPrefs, "PORT")
        FirstConnect = !sharedPrefs.contains("FirstConnect")

        var commuForm = CommuForm("SQL", "Select", "ALL", getMacAddr())
        pNettyTcpClient = setupNettyClient(ip, port, isheartBeat, commuForm)
        pNettyTcpClient.setListener(this)  //设置TCP监听


        GlobalScope.launch(Dispatchers.IO) {
            if (!pNettyTcpClient.connectStatus) {
                pNettyTcpClient.connect()
                delay(1000)
                Nettyconnected = pNettyTcpClient.connectStatus
                Log.d(TAG, " 已連接")
            } else {
                pNettyTcpClient.disconnect()
                Nettyconnected = false
                Log.d(TAG, " 已中斷連接")
            }
        }
    }

    fun getMacAddr(): String {
        try {
            val all: List<NetworkInterface> =
                Collections.list(NetworkInterface.getNetworkInterfaces())
            for (nif: NetworkInterface in all) {
                if (!nif.getName().equals("wlan0")) continue
                val macBytes: ByteArray = nif.getHardwareAddress() ?: return ""
                val res1 = StringBuilder()
                for (b in macBytes) {
                    res1.append(String.format("%02X:", b))
                }
                if (res1.length > 0) {
                    res1.deleteCharAt(res1.length - 1)
                }
                return res1.toString()
            }
        } catch (ex: Exception) {
        }
        return "02:00:00:00:00:00"
    }
